    def connect(self):
        max_retries = 3
        for attempt in range(max_retries):
            try:
                self.conn = sqlite3.connect(self.db_name, check_same_thread=False, timeout=20)
                self.conn.row_factory = sqlite3.Row
                self.cursor = self.conn.cursor()
                # Включаем поддержку внешних ключей
                self.cursor.execute("PRAGMA foreign_keys = ON")
                return
            except Exception as e:
                logger.error(f"❌ Ошибка подключения к БД (попытка {attempt + 1}): {e}")
                if attempt < max_retries - 1:
                    time.sleep(1)
                else:
                    raise

    def execute(self, query, params=None):
        max_retries = 3
        for attempt in range(max_retries):
            try:
                if params:
                    self.cursor.execute(query, params)
                else:
                    self.cursor.execute(query)
                self.conn.commit()
                return True
            except sqlite3.OperationalError as e:
                if "database is locked" in str(e) and attempt < max_retries - 1:
                    time.sleep(0.1 * (attempt + 1))
                    continue
                logger.error(f"❌ Ошибка БД: {e}")
                return False
            except Exception as e:
                logger.error(f"❌ Ошибка БД: {e}")
                return False
        return False

    def update_tap(self, user_id):
        """Обновление данных при тапе"""
        try:
            user = self.get_user(user_id)
            if not user:
                return 0, 0, 0

            # Проверяем лимит тапов
            today = date.today().isoformat()
            if user['last_tap_date'] != today:
                self.reset_daily_taps(user_id)
                daily_taps = 0
            else:
                daily_taps = user['daily_taps']

            if daily_taps >= MAX_DAILY_TAPS:
                return 0, daily_taps, MAX_DAILY_TAPS

            click_power = user['click_power']
            tap_value = TAP_BASE_VALUE * click_power
            new_robux = round(user['robux'] + tap_value, 2)
            new_taps = user['taps'] + 1
            new_daily_taps = daily_taps + 1

            self.cursor.execute('''
                UPDATE users 
                SET robux = ?, taps = ?, daily_taps = ?, last_tap_time = ?, last_tap_date = ?
                WHERE user_id = ?
            ''', (new_robux, new_taps, new_daily_taps, datetime.now().isoformat(), today, user_id))
            self.conn.commit()

            logger.info(f"✅ Тап: user={user_id}, +{tap_value:.2f}💰 (день {new_daily_taps}/{MAX_DAILY_TAPS})")
            return tap_value, new_daily_taps, MAX_DAILY_TAPS - new_daily_taps
        except Exception as e:
            logger.error(f"❌ Ошибка update_tap: {e}")
            return 0, 0, 0

    def can_tap(self, user_id):
        """Проверка, можно ли тапать"""
        try:
            user = self.get_user(user_id)
            if not user:
                return True, 0, 0

            # Проверка кулдауна
            if user.get('last_tap_time'):
                last_tap = datetime.fromisoformat(user['last_tap_time'])
                if datetime.now() - last_tap < timedelta(seconds=TAP_COOLDOWN):
                    return False, 0, 0

            # Проверка дневного лимита
            today = date.today().isoformat()
            if user['last_tap_date'] != today:
                return True, 0, MAX_DAILY_TAPS

            remaining = MAX_DAILY_TAPS - user['daily_taps']
            return remaining > 0, user['daily_taps'], remaining
        except Exception as e:
            logger.error(f"❌ Ошибка can_tap: {e}")
            return True, 0, MAX_DAILY_TAPS

    def create_duel(self, challenger_id, opponent_id, bet_amount, chat_id=None):
        try:
            self.cursor.execute('''
                SELECT * FROM active_duels 
                WHERE ((challenger_id = ? AND opponent_id = ?) OR (challenger_id = ? AND opponent_id = ?)) 
                AND status = 'pending'
            ''', (challenger_id, opponent_id, opponent_id, challenger_id))

            if self.cursor.fetchone():
                return False, "Уже есть активная дуэль между вами"

            now = datetime.now()
            expires = now + timedelta(minutes=2)

            self.cursor.execute('''
                INSERT INTO active_duels (challenger_id, opponent_id, bet_amount, status, created_at, expires_at, chat_id)
                VALUES (?, ?, ?, 'pending', ?, ?, ?)
            ''', (challenger_id, opponent_id, bet_amount, now.isoformat(), expires.isoformat(), chat_id))

            self.conn.commit()
            return True, "Дуэль создана"
        except Exception as e:
            logger.error(f"❌ Ошибка create_duel: {e}")
            return False, f"Ошибка: {e}"

def is_private(update: Update):
    return update.effective_chat.type == "private"


def is_group(update: Update):
    return update.effective_chat.type in ["group", "supergroup"]

# ==================== ГЛАВНОЕ МЕНЮ ====================

async def show_main_menu(update: Update, context: ContextTypes.DEFAULT_TYPE):
    if not is_private(update):
        await update.message.reply_text(
            "❌ Это меню доступно только в личных сообщениях с ботом!\nНапиши мне в личку: @roblox_clicker_bot"
        )
        return

    user = update.effective_user
    user_data = db.get_user(user.id)

    if user_data:
        robux = user_data['robux']
        taps = user_data['taps']
        daily_taps = user_data['daily_taps']
        click_power = user_data['click_power']
        auto_clickers = user_data['auto_clickers']
        wins = user_data['wins']
        losses = user_data['losses']
        casino_wins = user_data['casino_wins']
        casino_losses = user_data['casino_losses']
    else:
        robux = taps = daily_taps = wins = losses = casino_wins = casino_losses = 0
        click_power = 1
        auto_clickers = 0

    tap_value = TAP_BASE_VALUE * click_power
    win_rate = (wins / (wins + losses) * 100) if (wins + losses) > 0 else 0
    casino_rate = (casino_wins / (casino_wins + casino_losses) * 100) if (casino_wins + casino_losses) > 0 else 0
    remaining_taps = MAX_DAILY_TAPS - daily_taps

    text = (f"🎮 **ROBLOX CLICKER** 🎮\n\n"
            f"👤 **{user.first_name}**\n"
            f"💰 Робуксы: **{robux:.2f}**\n"
            f"👆 Тапов сегодня: **{daily_taps}/{MAX_DAILY_TAPS}**\n"
            f"⚡ Один тап: **{tap_value:.2f}💰** (x{click_power})\n"
            f"🤖 Автокликеров: **{auto_clickers}**\n"
            f"⚔️ Дуэли: {wins}W / {losses}L ({win_rate:.1f}%)\n"
            f"🎰 Казино: {casino_wins}W / {casino_losses}L ({casino_rate:.1f}%)\n"
            f"📊 Осталось тапов: {remaining_taps}")

    if update.callback_query:
        await update.callback_query.edit_message_text(text, parse_mode=ParseMode.MARKDOWN)
        await update.callback_query.message.reply_text("Выбери действие:", reply_markup=get_main_keyboard())
    else:
        await update.message.reply_text(text, reply_markup=get_main_keyboard(), parse_mode=ParseMode.MARKDOWN)

# ==================== ТОП ИГРОКОВ ====================

async def show_top(update: Update, context: ContextTypes.DEFAULT_TYPE):
    top_users = db.get_top_users(10)

    if not top_users:
        text = "🏆 **ТОП ИГРОКОВ** 🏆\n\nПока нет игроков в базе данных"
    else:
        text = "🏆 **ТОП-10 ИГРОКОВ** 🏆\n\n"

        for i, user in enumerate(top_users, 1):
            name = user['first_name'] or user['username'] or f"Игрок {user['user_id']}"

            if i == 1:
                medal = "🥇"
            elif i == 2:
                medal = "🥈"
            elif i == 3:
                medal = "🥉"
            else:
                medal = f"{i}."

            text += f"{medal} **{name}**\n"
            text += f"   💰 **{user['robux']:.2f}** робуксов\n"
            text += f"   👆 {user['taps']} тапов | ⚡ {user['click_power']}x\n"

            if user['wins'] > 0 or user['losses'] > 0:
                text += f"   ⚔️ {user['wins']}W-{user['losses']}L ({user['win_rate']}%)\n"

            text += "\n"

    keyboard = [[InlineKeyboardButton("🔄 Обновить", callback_data='refresh_top')]]

    await update.message.reply_text(text, reply_markup=InlineKeyboardMarkup(keyboard), parse_mode=ParseMode.MARKDOWN)


# ==================== КОМАНДА ВЫДАЧИ РОБУКСОВ ====================

async def give_command(update: Update, context: ContextTypes.DEFAULT_TYPE):
    user_id = update.effective_user.id

    if user_id != ADMIN_ID:
        await update.message.reply_text("⛔ Эта команда только для администратора!")
        return

    if len(context.args) < 2:
        await update.message.reply_text(
            "❌ Неправильный формат!\n"
            "Используй: `/give @username сумма`\n"
            "Или: `/give ID сумма`\n\n"
            "Примеры:\n"
            "`/give @durov 100`\n"
            "`/give 123456789 500`",
            parse_mode=ParseMode.MARKDOWN
        )
        return

    target = context.args[0]
    try:
        amount = float(context.args[1])
    except ValueError:
        await update.message.reply_text("❌ Сумма должна быть числом!")
        return

    if amount <= 0:
        await update.message.reply_text("❌ Сумма должна быть положительной!")
        return

    if amount > 1000000:
        await update.message.reply_text("❌ Максимальная сумма: 1,000,000💰")
        return

    success, message = db.add_robux(user_id, target, amount)

    if success:
        await update.message.reply_text(message)

        try:
            user = db.get_user_by_name_or_id(target)
            if user:
                await context.bot.send_message(
                    user['user_id'],
                    f"💰 Админ выдал тебе {amount} робуксов!"
                )
        except:
            pass
    else:
        await update.message.reply_text(message)

# ==================== ТАП ====================

async def handle_tap(update: Update, context: ContextTypes.DEFAULT_TYPE):
    if not is_private(update):
        await update.message.reply_text(
            "❌ Тапать можно только в личных сообщениях с ботом!\nНапиши мне в личку: @roblox_clicker_bot"
        )
        return

    user = update.effective_user

    can_tap, current, remaining = db.can_tap(user.id)

    if not can_tap:
        if current >= MAX_DAILY_TAPS:
            await update.message.reply_text(
                f"⏳ Ты исчерпал лимит на сегодня ({MAX_DAILY_TAPS} тапов)! Завтра продолжим 😊"
            )
        else:
            await update.message.reply_text("⏳ Подожди секунду между тапами!")
        return

    tap_value, new_daily, remaining = db.update_tap(user.id)

    if tap_value > 0:
        await update.message.reply_text(
            f"✅ +{tap_value:.2f}💰\n"
            f"📊 Сегодня: {new_daily}/{MAX_DAILY_TAPS}\n"
            f"⏳ Осталось: {remaining} тапов"
        )
        await show_main_menu(update, context)
    else:
        await update.message.reply_text("❌ Ошибка при тапе")

async def casino_choose_bet(update: Update, context: ContextTypes.DEFAULT_TYPE):
    query = update.callback_query
    user = query.from_user
    data = query.data

    user_data = db.get_user(user.id)
    robux = user_data['robux'] if user_data else 0

    if data == 'casino_custom':
        text = (f"🎰 **ВВЕДИ СТАВКУ**\n\n"
                f"💰 Баланс: {robux:.2f}\n"
                f"📊 От {CASINO_MIN_BET} до {CASINO_MAX_BET}💰\n"
                f"🎲 Шанс: {CASINO_WIN_RATE}%\n\n"
                f"Отправь число:")

        keyboard = [[InlineKeyboardButton("◀ Отмена", callback_data='casino_cancel')]]
        context.user_data['casino_mode'] = 'waiting_bet'

        await query.edit_message_text(text, reply_markup=InlineKeyboardMarkup(keyboard), parse_mode=ParseMode.MARKDOWN)
    else:
        bet = int(data.split('_')[1])

        if bet > robux:
            await query.answer(f"❌ У тебя только {robux:.2f}💰!", show_alert=True)
            return

        if bet < CASINO_MIN_BET or bet > CASINO_MAX_BET:
            await query.answer(f"❌ От {CASINO_MIN_BET} до {CASINO_MAX_BET}💰!", show_alert=True)
            return

        context.user_data['casino_bet'] = bet
        await casino_choose_side(update, context)

        user_data = db.get_user(user.id)
        robux = user_data['robux'] if user_data else 0

            await update.message.reply_text(f"❌ От {CASINO_MIN_BET} до {CASINO_MAX_BET}💰!")

                f"💰 Ставка: {bet}💰\n"
                f"🎲 Шанс: {CASINO_WIN_RATE}%")

        keyboard = [
            [InlineKeyboardButton("🪙 Орёл", callback_data='casino_heads'),
             InlineKeyboardButton("🪙 Решка", callback_data='casino_tails')],

async def duel_command(update: Update, context: ContextTypes.DEFAULT_TYPE):
    user = update.effective_user
    chat = update.effective_chat
    message = update.message

    if len(context.args) != 2:
        await update.message.reply_text(
            "❌ **Неправильный формат!**\n\n"
            "Используй: `/duel @username сумма`\n"
            "Или: `/duel ID сумма`\n\n"
            "Пример: `/duel @durov 100`",
            parse_mode=ParseMode.MARKDOWN
        )
        return

    try:
        target = context.args[0]
        bet = int(context.args[1])

        if bet < DUEL_BET_MIN or bet > DUEL_BET_MAX:
            await update.message.reply_text(f"❌ Ставка должна быть от {DUEL_BET_MIN} до {DUEL_BET_MAX}💰")
            return

        challenger = db.get_user(user.id)
        if not challenger or challenger['robux'] < bet:
            await update.message.reply_text(f"❌ У тебя недостаточно робуксов! Нужно {bet}💰")
            return

        opponent = None
        opponent_user = None
        found_by = ""

        if target.startswith('@'):
            opponent = db.get_user_by_username(target)
            if opponent:
                found_by = "БД"
        else:
            try:
                user_id = int(target)
                opponent = db.get_user(user_id)
                if opponent:
                    found_by = "БД"
            except ValueError:
                pass

        if not opponent and is_group(update):
            try:
                if target.startswith('@'):
                    username = target[1:]
                    member = await context.bot.get_chat_member(chat.id, username)
                else:
                    member = await context.bot.get_chat_member(chat.id, int(target))

                if member and member.user:
                    db.add_user(member.user.id, member.user.username, member.user.first_name)
                    opponent = db.get_user(member.user.id)
                    opponent_user = member.user
                    found_by = "чат"
            except:
                pass

        if not opponent:
            await update.message.reply_text(
                "❌ **Противник не найден!**\n\n"
                "Попроси противника написать /start в боте",
                parse_mode=ParseMode.MARKDOWN
            )
            return

        opponent_id = opponent['user_id']

        if user.id == opponent_id:
            await update.message.reply_text("❌ Нельзя вызвать самого себя!")
            return

        if opponent['robux'] < bet:
            await update.message.reply_text("❌ У противника недостаточно робуксов!")
            return

        if db.get_pending_duel(user.id):
            await update.message.reply_text("❌ У тебя уже есть активная дуэль!")
            return

        if db.get_pending_duel(opponent_id):
            await update.message.reply_text("❌ У противника уже есть активная дуэль!")
            return

        success, msg = db.create_duel(user.id, opponent_id, bet, chat.id)

        if success:
            duel = db.get_pending_duel(user.id)
            duel_id = duel['duel_id'] if duel else 0

            opponent_name = opponent['first_name'] or opponent['username'] or str(opponent_id)
            await update.message.reply_text(
                f"✅ **Вызов отправлен!**\n\n"
                f"👤 Противник: {opponent_name}\n"
                f"💰 Ставка: {bet}💰\n"
                f"📍 Найден в: {found_by}\n\n"
                f"⏳ Ожидаем ответа...",
                parse_mode=ParseMode.MARKDOWN
            )

            keyboard = [[
                InlineKeyboardButton("✅ ПРИНЯТЬ ДУЭЛЬ", callback_data=f'accept_duel_{duel_id}'),
                InlineKeyboardButton("❌ ОТКАЗАТЬСЯ", callback_data=f'reject_duel_{duel_id}')
            ]]

            try:
                await context.bot.send_message(
                    opponent_id,
                    f"⚔️ **ТЕБЯ ВЫЗВАЛИ НА ДУЭЛЬ!** ⚔️\n\n"
                    f"👤 Противник: {user.first_name}\n"
                    f"💰 Ставка: {bet}💰\n\n"
                    f"**Нажми кнопку чтобы ответить:**",
                    reply_markup=InlineKeyboardMarkup(keyboard),
                    parse_mode=ParseMode.MARKDOWN
                )
            except:
                if is_group(update):
                    mention = ""
                    if opponent_user and opponent_user.username:
                        mention = f"@{opponent_user.username}"
                    elif opponent_user:
                        mention = f"[{opponent_user.first_name}](tg://user?id={opponent_id})"
                    else:
                        mention = opponent_name

                    await update.message.reply_text(
                        f"⚠️ Не удалось отправить личное сообщение.\n"
                        f"{mention}, **прими вызов здесь:**",
                        reply_markup=InlineKeyboardMarkup(keyboard),
                        parse_mode=ParseMode.MARKDOWN
                    )
        else:
            await update.message.reply_text(f"❌ {msg}")

    except ValueError:
        await update.message.reply_text("❌ Сумма должна быть числом!")
    except Exception as e:
        logger.error(f"Ошибка в duel_command: {e}")
        await update.message.reply_text("❌ Произошла ошибка")


async def accept_duel(update: Update, context: ContextTypes.DEFAULT_TYPE):
    query = update.callback_query
    user = query.from_user
    duel_id = int(query.data.split('_')[2])

    await query.answer()

    duel = db.get_duel_by_id(duel_id)
    if not duel:
        await query.edit_message_text("❌ Дуэль не найдена или истекла")
        return

    if user.id != duel['opponent_id']:
        await query.edit_message_text("⛔ Это не твоя дуэль!")
        return

    challenger = db.get_user(duel['challenger_id'])
    opponent = db.get_user(duel['opponent_id'])

    if not challenger or challenger['robux'] < duel['bet_amount']:
        await query.edit_message_text("❌ У противника недостаточно робуксов! Дуэль отменена.")
        db.cancel_duel(duel_id)
        return

    if not opponent or opponent['robux'] < duel['bet_amount']:
        await query.edit_message_text("❌ У тебя недостаточно робуксов! Дуэль отменена.")
        db.cancel_duel(duel_id)
        return

    winner_id = random.choice([duel['challenger_id'], duel['opponent_id']])
    loser_id = duel['opponent_id'] if winner_id == duel['challenger_id'] else duel['challenger_id']

    success, bet = db.resolve_duel(duel_id, winner_id, loser_id)

    if success:
        try:
            winner = await context.bot.get_chat(winner_id)
            loser = await context.bot.get_chat(loser_id)
            winner_name = winner.first_name
            loser_name = loser.first_name
        except:
            winner_name = f"ID {winner_id}"
            loser_name = f"ID {loser_id}"

        result_text = (f"⚔️ **ДУЭЛЬ ЗАВЕРШЕНА!** ⚔️\n\n"
                       f"🏆 **Победитель:** {winner_name}\n"
                       f"💔 **Проигравший:** {loser_name}\n"
                       f"💰 **Выигрыш:** {bet}💰")

        await query.edit_message_text(result_text, parse_mode=ParseMode.MARKDOWN)

        try:
            await context.bot.send_message(winner_id, f"🎉 Ты победил в дуэли и выиграл {bet}💰!")
            await context.bot.send_message(loser_id, f"😢 Ты проиграл в дуэли и потерял {bet}💰")
        except:
            pass
    else:
        await query.edit_message_text("❌ Ошибка при проведении дуэли")

async def show_stats(update: Update, context: ContextTypes.DEFAULT_TYPE):

    user_data = db.get_user(user.id)

    if user_data:
        joined = datetime.fromisoformat(user_data['joined_date']).strftime("%d.%m.%Y")
        wins = user_data['wins']
        losses = user_data['losses']
        casino_wins = user_data['casino_wins']
        casino_losses = user_data['casino_losses']
        win_rate = (wins / (wins + losses) * 100) if (wins + losses) > 0 else 0
        casino_rate = (casino_wins / (casino_wins + casino_losses) * 100) if (casino_wins + casino_losses) > 0 else 0

        text = (f"📊 **ТВОЯ СТАТИСТИКА**\n\n"
                f"👤 Игрок: {user.first_name}\n"
                f"🆔 ID: `{user.id}`\n"
                f"📅 С: {joined}\n"
                f"💰 Робуксы: {user_data['robux']:.2f}\n"
                f"👆 Тапов: {user_data['taps']}\n"
                f"⚡ Сила: {user_data['click_power']}x\n"
                f"🤖 Автокликеров: {user_data['auto_clickers']}\n"
                f"⚔️ Дуэли: {wins}W / {losses}L ({win_rate:.1f}%)\n"
                f"🎰 Казино: {casino_wins}W / {casino_losses}L ({casino_rate:.1f}%)")
    else:
        text = "❌ Нет данных"

    await update.message.reply_text(text, parse_mode=ParseMode.MARKDOWN)

async def auto_clicker_job(context: ContextTypes.DEFAULT_TYPE):
    try:
        db.cursor.execute("SELECT user_id, auto_clickers FROM users WHERE auto_clickers > 0")
        users = db.cursor.fetchall()

        for user_id, clicks in users:
            db.cursor.execute("UPDATE users SET robux = robux + ? WHERE user_id = ?", (clicks, user_id))

        db.conn.commit()
        if users:
            logger.info(f"🔄 Автокликер: {len(users)} пользователей")
    except Exception as e:
        logger.error(f"❌ Ошибка автокликера: {e}")

async def button_handler(update: Update, context: ContextTypes.DEFAULT_TYPE):
    query = update.callback_query
    data = query.data

    try:
        if data == 'back_to_menu':
            await query.answer()
            if is_private(update):
                fake_update = update
                fake_update.message = query.message
                await show_main_menu(fake_update, context)
            else:
                await query.edit_message_text("❌ Это меню доступно только в личных сообщениях!")

        elif data == 'refresh_top':
            await query.answer()
            await show_top(update, context)

        elif data.startswith('casino_'):
            if data in ['casino_cancel', 'casino_back']:
                if data == 'casino_cancel':
                    await casino_cancel(update, context)
                else:
                    await casino_back(update, context)
            elif data == 'casino_play_again':
                await casino_play_again(update, context)
            elif data in ['casino_10', 'casino_50', 'casino_100', 'casino_500', 'casino_1000', 'casino_5000',
                          'casino_custom']:
                await casino_choose_bet(update, context)
            elif data in ['casino_heads', 'casino_tails']:
                await casino_play(update, context)

        elif data.startswith('buy_'):
            if not is_private(update):
                await query.answer("❌ Магазин только в личке!", show_alert=True)
                return
            await handle_buy(update, context)

        elif data.startswith('accept_duel_'):
            await accept_duel(update, context)
        elif data.startswith('reject_duel_'):
            await reject_duel(update, context)
        elif data.startswith('cancel_duel_'):
            await cancel_duel(update, context)

    except Exception as e:
        logger.error(f"Ошибка в button_handler: {e}")
        await query.answer("❌ Произошла ошибка", show_alert=True)

async def run_bot_with_restart():
    max_restarts = 10
    restart_count = 0

    while restart_count < max_restarts:
        try:
            logger.info(f"🚀 Запуск бота (попытка {restart_count + 1})...")

            app = Application.builder().token(BOT_TOKEN).build()

            app.add_handler(CommandHandler("start", start))
            app.add_handler(CommandHandler("admin", admin_command))
            app.add_handler(CommandHandler("duel", duel_command))
            app.add_handler(CommandHandler("give", give_command))
            app.add_handler(CommandHandler("top", show_top))

            app.add_handler(MessageHandler(filters.TEXT & ~filters.COMMAND, text_handler))
            app.add_handler(CallbackQueryHandler(button_handler))

            # Добавляем авто-бэкап каждый час
            if app.job_queue:
                app.job_queue.run_repeating(auto_clicker_job, interval=10, first=10)
                app.job_queue.run_repeating(auto_backup_job, interval=3600, first=3600)  # Каждый час

            logger.info("✅ Бот запущен!")
            await app.initialize()
            await app.start()
            await app.updater.start_polling()

            while True:
                await asyncio.sleep(1)

        except (NetworkError, TimedOut) as e:
            logger.error(f"🌐 Сетевая ошибка: {e}")
            restart_count += 1
            logger.info(f"🔄 Перезапуск через 5 секунд...")
            await asyncio.sleep(5)

        except Exception as e:
            logger.error(f"❌ Критическая ошибка: {e}")
            logger.error(traceback.format_exc())
            restart_count += 1
            logger.info(f"🔄 Перезапуск через 10 секунд...")
            await asyncio.sleep(10)

        finally:
            try:
                await app.updater.stop()
                await app.stop()
                await app.shutdown()
            except:
                pass

    logger.critical("💀 Превышено количество перезапусков. Бот остановлен.")


# ==================== ЗАПУСК ====================

def main():
    try:
        asyncio.run(run_bot_with_restart())
    except KeyboardInterrupt:
        logger.info("🛑 Бот остановлен пользователем")
    except Exception as e:
        logger.critical(f"💥 Фатальная ошибка: {e}")
        logger.critical(traceback.format_exc())


if __name__ == "__main__":
    main()
